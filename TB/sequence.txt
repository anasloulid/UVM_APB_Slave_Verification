class APB_SEQUENCE extends uvm_sequence;
  `umv_object_utils(APB_SEQUENCE);
  
  function new(string name = "APB_SEQUENCE");
    super.new(name);
  endfunction
  //transaction handle
  APB_TRANS trans;
  
  //task body
  task body();
    trans = APB_TRANS::type_id::create("trans");
    start_item(trans);
      // Randomize transaction
      if (!trans.randomize())
        `uvm_fatal(get_type_name(), "Randomization Failure")
     finish_item(trans); 
    trans.print();
  endtask
endclass
  
  
  
  
  
  
  
  
  
  
endclass
